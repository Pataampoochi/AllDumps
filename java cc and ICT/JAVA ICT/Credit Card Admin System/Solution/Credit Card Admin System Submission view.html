   24 	public static List<CreditCard> buildMasterCreditCardList(List<String> records) {

   26 		for (String record : records) {

   35 				card.setBillAmount(getBillAmountWithLatePaymentCharges(Double.parseDouble(cardElements[4])));

   37 				card.setBillAmount(Double.parseDouble(cardElements[4]));
   38 			}

   43 			
   44 			masterCreditCards.add(card);
